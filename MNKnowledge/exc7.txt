func singleElement(Array of int a, int c):
	Let s := Array of (c+1) zeroes
	
	for each int i in a:
		s[c] := 0
		Let p := 0
		Let q := 0
		
		for k from c down to 1:
			s[k] := s[k] XOR p
			p := s[k-1] AND i
			s[k] := s[k] OR p
			q := q OR s[k]
		
		s[0] = s[0] XOR p
		s[0] = s[0] OR (NOT(q) AND i)
		s[0] = s[0] XOR s[c]
		
	return s[0]
